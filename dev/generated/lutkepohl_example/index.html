Take the subset from 1960q1 to 1978q4, and use log-difference for each series

julia> Y = @linq datadf |>
           where(:date .< Date(1979,3,1)) |>
           select(investment = diff(log.(:invest)), income = diff(log.(:income)),
               consumption = diff(log.(:cons)))
